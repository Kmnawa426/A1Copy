Description

Problem 5 : Write a Program in C# to check a number if it is Prime; otherwise display the factor of that number.

Solution

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp_PartA_Problem5
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int num, i;
            Boolean isPrime = true;
            Console.WriteLine("Enter the Number:");
            num = int.Parse(Console.ReadLine());

            Console.WriteLine("Factors of " + num + ":");
            for (i = 2; i < num; i++)
            {
                if (num % i == 0)
                { 
                    isPrime = false;
                    Console.WriteLine(i);
                }
            }
            if (isPrime)
            {
                Console.WriteLine(num);
                Console.WriteLine("It is a Prime Number:");
            }
            Console.ReadKey();
        }
    }
}
